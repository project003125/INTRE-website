def mkt_equations(psi, t, params, external_stress=0.0):
    """
    鐘舵€佹紨鍖栧井鍒嗘柟绋?
    psi: 鐘舵€佸悜閲忥紙褰撳墠鍊硷級
    t: 鏃堕棿锛堝皬鏃讹級
    params: 鍙傛暟瀛楀吀
    external_stress: 澶栭儴鍘嬪姏锛?-100锛?
    """
    dpsi = np.zeros_like(psi)
    
    # 1. 璧勬簮鍔ㄥ姏瀛︼細M3-RES-D锛堟剰蹇楄祫婧愶級
    # dR/dt = supply - alpha * R - demand
    R3 = psi[STATE_INDICES['M3_RES_D']]
    supply_m3 = params.get('supply_m3', 15.0)   # 鏉ヨ嚜M1鐨勫熀纭€渚涚粰锛?h锛?
    alpha = params.get('alpha_m3', 0.12)        # 娑堣€楅€熺巼甯告暟
    # 闇€姹傦細鏉ヨ嚜浠诲姟鍘嬪姏 + M4-DSP-B鐨勬秷鑰?
    demand = params.get('base_demand', 5.0) + external_stress * 0.1
    demand += psi[STATE_INDICES['M4_DSP_B']] * 0.15
    dpsi[STATE_INDICES['M3_RES_D']] = supply_m3 - alpha * R3 - demand
    
    # 2. 璧勬簮鍔ㄥ姏瀛︼細M4-RES-D锛堟儏缁闄愶級
    R4 = psi[STATE_INDICES['M4_RES_D']]
    supply_m4 = params.get('supply_m4', 10.0)
    beta = params.get('beta_m4', 0.10)
    consumption_m4 = psi[STATE_INDICES['M4_DSP_B']] * 0.2
    dpsi[STATE_INDICES['M4_RES_D']] = supply_m4 - beta * R4 - consumption_m4
    
    # 3. 璋冨害鍔ㄥ姏瀛︼細M4-DSP-B锛堣嚜鍔ㄦ儏缁縺娲伙級鐨勯槇鍊兼ā鍨?
    B4 = psi[STATE_INDICES['M4_DSP_B']]
    # 婵€娲婚┍鍔ㄥ姏锛氬▉鑳佷俊蹇?M4_STR_R) + 澶栭儴鍘嬪姏
    drive = psi[STATE_INDICES['M4_STR_R']] * 50.0 + external_stress * 0.5
    # 婵€娲婚€熺巼闅忚祫婧愪笅闄嶈€屽崌楂橈紙鍘绘姂鍒讹級
    inhibition = 1.0 / (1.0 + np.exp(-(R3 - 40.0) / 10.0))  # 璧勬簮楂樹簬40鏃舵姂鍒跺己
    activation_rate = params.get('activation_rate', 0.3) * drive * (1 - inhibition)
    # 鑷劧鎭㈠锛堝悜鍩虹嚎鍥炲綊锛?
    recovery_rate = params.get('recovery_rate_b4', 0.1)
    target_b4 = 30.0  # 鍋ュ悍鍩虹嚎
    dpsi[STATE_INDICES['M4_DSP_B']] = activation_rate - recovery_rate * (B4 - target_b4)
    
    # 4. 缁撴瀯鍔ㄥ姏瀛︼細M4_STR_R锛堝▉鑳佷俊蹇佃鍒欙級缂撴參鏇存柊
    S4 = psi[STATE_INDICES['M4_STR_R']]
    learning = params.get('learning_rate', 0.01) * (B4 / 100.0) * (external_stress / 100.0)
    decay = 0.002
    dpsi[STATE_INDICES['M4_STR_R']] = learning * (1 - S4) - decay * S4
    dpsi[STATE_INDICES['M4_STR_R']] = np.clip(dpsi[STATE_INDICES['M4_STR_R']], -0.5, 0.5)
    
    return dpsi

# 绮掑瓙婊ゆ尝锛圔ootstrap Filter锛?
def particle_filter(observations, t_eval, n_particles=500, process_noise_std=2.0):
    particles = np.random.uniform(20, 80, size=n_particles)
    weights = np.ones(n_particles) / n_particles
    estimated_states = []
    for t_idx, y in enumerate(observations):
        # 棰勬祴姝ラ
        particles = particles + np.random.normal(0, process_noise_std, size=n_particles)
        particles = np.clip(particles, 0, 100)
        # 鏇存柊鏉冮噸
        likelihood = np.exp(-0.5 * ((particles - y) / 5.0)**2)
        weights = likelihood * weights
        weights += 1e-300
        weights /= np.sum(weights)
        est = np.sum(particles * weights)
        estimated_states.append(est)
        # 閲嶉噰鏍?
        if 1 / np.sum(weights**2) < n_particles / 2:
            indices = np.random.choice(n_particles, size=n_particles, p=weights)
            particles = particles[indices]
            weights = np.ones(n_particles) / n_particles
    return np.array(estimated_states)
