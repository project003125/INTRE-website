附录C：PSE快速入门——代码示例与仿真模板

本附录提供PSE（Psychological System Engine）开源引擎的快速上手指南。PSE是用Python实现的仿真框架，用于运行M-R-D-S状态演化、蒙特卡罗智能体仿真和贝叶斯校准。本附录面向有一定Python基础的研究者和开发者，提供从安装、基本状态定义、运行仿真到可视化的最小工作示例。

环境要求：Python 3.9+，建议使用Anaconda或Miniconda管理环境。推荐安装以下依赖库：

  • numpy（数值计算）
  • scipy（微分方程求解、统计分布）
  • matplotlib（可视化）
  • pandas（数据处理）
  • numba（可选，加速蒙特卡罗仿真）
  • jax（可选，自动微分与GPU加速）

完整代码库及文档请访问：https://github.com/INTRE/PSE（示例地址）

C.2 定义M-R-D-S状态空间

PSE的核心数据结构是42维状态向量Psi。本示例定义状态向量的索引映射和初始状态。


# 定义子维度索引（简化版，仅列出部分核心维度）
STATE_INDICES = {
    # M1 具身
    'M1_RES_D': 0,
    'M1_RES_B': 1,
    'M1_DSP_F': 2,
    'M1_DSP_B': 3,
    # M2 认知
    'M2_RES_D': 4,
    'M2_RES_B': 5,
    'M2_DSP_F': 6,
    'M2_DSP_B': 7,
    'M2_STR_R': 8,   # 信念规则强度（0-1）
    # M3 意志
    'M3_RES_D': 9,
    'M3_RES_B': 10,
    'M3_DSP_F': 11,
    'M3_DSP_B': 12,
    'M3_STR_R': 13,
    # M4 情绪
    'M4_RES_D': 14,
    'M4_RES_B': 15,
    'M4_DSP_F': 16,
    'M4_DSP_B': 17,
    'M4_STR_R': 18,
    'M4_STR_C': 19,   # 情感连接权重（0-1）
    # M5 言语（简化）
    'M5_RES_D': 20,
    'M5_STR_R': 21,
    # M6 行为
    'M6_RES_D': 22,
    'M6_DSP_F': 23,
    'M6_DSP_B': 24,
    'M6_STR_S': 25,   # 习惯序列强度
}
DIM = len(STATE_INDICES)  # 26维（完整42维请使用预定义常量）

# 创建初始状态向量（健康基线）
def initial_healthy_state():
    psi = np.zeros(DIM)
    # 设置默认值（0-100尺度，信念/连接为0-1）
    psi[STATE_INDICES['M1_RES_D']] = 75.0
    psi[STATE_INDICES['M2_RES_D']] = 80.0
    psi[STATE_INDICES['M3_RES_D']] = 70.0
    psi[STATE_INDICES['M3_DSP_F']] = 80.0
    psi[STATE_INDICES['M4_RES_D']] = 75.0
    psi[STATE_INDICES['M4_RES_B']] = 70.0
    psi[STATE_INDICES['M4_DSP_B']] = 30.0   # 低背景焦虑
    psi[STATE_INDICES['M4_STR_R']] = 0.3    # 健康的威胁评估
    psi[STATE_INDICES['M6_RES_D']] = 70.0
    psi[STATE_INDICES['M6_DSP_F']] = 75.0
    return psi

# 创建焦虑状态（M4-DSP-B高，资源低）
def initial_anxious_state():
    psi = initial_healthy_state()
    psi[STATE_INDICES['M4_DSP_B']] = 85.0
    psi[STATE_INDICES['M3_RES_D']] = 45.0
    psi[STATE_INDICES['M4_RES_D']] = 40.0
    psi[STATE_INDICES['M4_STR_R']] = 0.75
    return psi

C.3 定义动力学方程（MKT简化版）

本示例实现MKT资源动力学和调度切换的简化版本。


def mkt_equations(psi, t, params, external_stress=0.0):
    """
    状态演化微分方程
    psi: 状态向量（当前值）
    t: 时间（小时）
    params: 参数字典
    external_stress: 外部压力（0-100）
    """
    dpsi = np.zeros_like(psi)
    
    # 1. 资源动力学：M3-RES-D（意志资源）
    # dR/dt = supply - alpha * R - demand
    R3 = psi[STATE_INDICES['M3_RES_D']]
    supply_m3 = params.get('supply_m3', 15.0)   # 来自M1的基础供给（/h）
    alpha = params.get('alpha_m3', 0.12)        # 消耗速率常数
    # 需求：来自任务压力 + M4-DSP-B的消耗
    demand = params.get('base_demand', 5.0) + external_stress * 0.1
    demand += psi[STATE_INDICES['M4_DSP_B']] * 0.15
    dpsi[STATE_INDICES['M3_RES_D']] = supply_m3 - alpha * R3 - demand
    
    # 2. 资源动力学：M4-RES-D（情绪容限）
    R4 = psi[STATE_INDICES['M4_RES_D']]
    supply_m4 = params.get('supply_m4', 10.0)
    beta = params.get('beta_m4', 0.10)
    consumption_m4 = psi[STATE_INDICES['M4_DSP_B']] * 0.2
    dpsi[STATE_INDICES['M4_RES_D']] = supply_m4 - beta * R4 - consumption_m4
    
    # 3. 调度动力学：M4-DSP-B（自动情绪激活）的阈值模型
    B4 = psi[STATE_INDICES['M4_DSP_B']]
    # 激活驱动力：威胁信念(M4_STR_R) + 外部压力
    drive = psi[STATE_INDICES['M4_STR_R']] * 50.0 + external_stress * 0.5
    # 激活速率随资源下降而升高（去抑制）
    inhibition = 1.0 / (1.0 + np.exp(-(R3 - 40.0) / 10.0))  # 资源高于40时抑制强
    activation_rate = params.get('activation_rate', 0.3) * drive * (1 - inhibition)
    # 自然恢复（向基线回归）
    recovery_rate = params.get('recovery_rate_b4', 0.1)
    target_b4 = 30.0  # 健康基线
    dpsi[STATE_INDICES['M4_DSP_B']] = activation_rate - recovery_rate * (B4 - target_b4)
    
    # 4. 结构动力学：M4_STR_R（威胁信念规则）缓慢更新
    S4 = psi[STATE_INDICES['M4_STR_R']]
    learning = params.get('learning_rate', 0.01) * (B4 / 100.0) * (external_stress / 100.0)
    decay = 0.002
    dpsi[STATE_INDICES['M4_STR_R']] = learning * (1 - S4) - decay * S4
    dpsi[STATE_INDICES['M4_STR_R']] = np.clip(dpsi[STATE_INDICES['M4_STR_R']], -0.5, 0.5)
    
    return dpsi

C.4 运行仿真与可视化


# 设置仿真参数
t_max = 72  # 小时
dt = 0.1
t_eval = np.arange(0, t_max, dt)

# 定义外部压力序列（例如：工作压力有日节律）
def external_stress_profile(t):
    hour = t % 24
    if 9 <= hour < 18:
        stress = 60.0 + 20 * np.sin((hour - 9) / 9 * np.pi)  # 峰值80
    else:
        stress = 10.0
    return stress

# 准备求解
psi0 = initial_anxious_state()
params = {
    'supply_m3': 15.0,
    'alpha_m3': 0.12,
    'base_demand': 5.0,
    'supply_m4': 10.0,
    'beta_m4': 0.10,
    'activation_rate': 0.3,
    'recovery_rate_b4': 0.1,
    'learning_rate': 0.01,
}

# 将外部压力作为时间的函数传递
def psi_derivative_wrapper(psi, t, params):
    stress = external_stress_profile(t)
    return mkt_equations(psi, t, params, external_stress=stress)

# 求解ODE
from scipy.integrate import odeint
sol = odeint(psi_derivative_wrapper, psi0, t_eval, args=(params,))

# 提取关键变量
M3_RES_D = sol[:, STATE_INDICES['M3_RES_D']]
M4_DSP_B = sol[:, STATE_INDICES['M4_DSP_B']]
M4_STR_R = sol[:, STATE_INDICES['M4_STR_R']]

# 绘制
fig, axes = plt.subplots(3, 1, figsize=(10, 8), sharex=True)
axes[0].plot(t_eval, M3_RES_D, label='M3-RES-D (意志资源)', color='blue')
axes[0].axhline(y=30, color='r', linestyle='--', label='功能阈值 (30)')
axes[0].set_ylabel('资源水平 (0-100)')

axes[1].plot(t_eval, M4_DSP_B, label='M4-DSP-B (自动情绪激活)', color='orange')
axes[1].axhline(y=70, color='r', linestyle='--', label='高激活阈值 (70)')
axes[1].set_ylabel('激活强度 (0-100)')
axes[1].legend()
axes[1].grid(True)

axes[2].plot(t_eval, M4_STR_R, label='M4-STR-R (威胁信念权重)', color='green')
axes[2].set_xlabel('时间 (小时)')
axes[2].set_ylabel('信念权重 (0-1)')
axes[2].legend()
axes[2].grid(True)

C.5 蒙特卡罗仿真（PSE-M）示例

生成1000个虚拟个体，参数从先验分布中采样，模拟群体变异性。


# 参数先验分布
def sample_parameters(n_samples):
    alphas = np.random.gamma(shape=12, scale=0.01, size=n_samples)  # mean=0.12
    betas = np.random.gamma(shape=10, scale=0.01, size=n_samples)   # mean=0.10
    return {'alpha_m3': alphas, 'beta_m4': betas}

# 蒙特卡罗仿真函数
def monte_carlo_simulation(n_samples, t_max=72, dt=0.1):
    t_eval = np.arange(0, t_max, dt)
    param_samples = sample_parameters(n_samples)
    trajectories = []
    for i in range(n_samples):
        params_i = {'alpha_m3': param_samples['alpha_m3'][i],
                    'beta_m4': param_samples['beta_m4'][i],
                    'supply_m3': 15.0, 'base_demand': 5.0, 'supply_m4': 10.0,
                    'activation_rate': 0.3, 'recovery_rate_b4': 0.1, 'learning_rate': 0.01}
        psi0 = initial_anxious_state()
        sol = odeint(psi_derivative_wrapper, psi0, t_eval, args=(params_i,))
        trajectories.append(sol[:, STATE_INDICES['M3_RES_D']])
    return np.array(trajectories)

# 运行（注意：n_samples较大时耗时较长，可使用numba加速）
traj_m3 = monte_carlo_simulation(200)  # 200个样本演示

# 计算分位数
p2_5 = np.percentile(traj_m3, 2.5, axis=0)
p50 = np.percentile(traj_m3, 50, axis=0)
p97_5 = np.percentile(traj_m3, 97.5, axis=0)

plt.figure(figsize=(10, 5))
plt.fill_between(t_eval, p2_5, p97_5, alpha=0.3, label='95%预测区间')
plt.plot(t_eval, p50, 'b-', label='中位数轨迹')
plt.axhline(y=30, color='r', linestyle='--', label='功能阈值')
plt.xlabel('时间 (小时)')
plt.ylabel('M3-RES-D (意志资源)')
plt.title('蒙特卡罗仿真：资源消耗轨迹的预测区间')
plt.legend()
plt.grid(True)
plt.show()

C.6 简单贝叶斯校准（PSE-C）示例

使用粒子滤波从含噪观测中估计个体的M3-RES-D状态。


# 生成模拟观测数据（从真实状态中加入噪声）
def generate_observations(true_state, t_eval, noise_std=5.0):
    obs = true_state + np.random.normal(0, noise_std, size=len(true_state))
    return np.clip(obs, 0, 100)

# 粒子滤波（Bootstrap Filter）
def particle_filter(observations, t_eval, n_particles=500, process_noise_std=2.0):
    particles = np.random.uniform(20, 80, size=n_particles)
    weights = np.ones(n_particles) / n_particles
    estimated_states = []
    for t_idx, y in enumerate(observations):
        # 预测步骤
        particles = particles + np.random.normal(0, process_noise_std, size=n_particles)
        particles = np.clip(particles, 0, 100)
        # 更新权重
        likelihood = np.exp(-0.5 * ((particles - y) / 5.0)**2)
        weights = likelihood * weights
        weights += 1e-300
        weights /= np.sum(weights)
        est = np.sum(particles * weights)
        estimated_states.append(est)
        # 重采样
        if 1 / np.sum(weights**2) < n_particles / 2:
            indices = np.random.choice(n_particles, size=n_particles, p=weights)
            particles = particles[indices]
            weights = np.ones(n_particles) / n_particles
    return np.array(estimated_states)

# 运行示例（使用之前仿真的真实轨迹）
true_traj = sol[:, STATE_INDICES['M3_RES_D']]  # 之前仿真的轨迹
obs = generate_observations(true_traj, t_eval, noise_std=6.0)
filtered = particle_filter(obs, t_eval, n_particles=500)

plt.figure(figsize=(10, 5))
plt.plot(t_eval, true_traj, 'g-', label='真实状态（未知）')
plt.scatter(t_eval[::10], obs[::10], color='red', s=10, alpha=0.5, label='含噪观测')
plt.plot(t_eval, filtered, 'b--', label='粒子滤波估计')
plt.xlabel('时间 (小时)')
plt.ylabel('M3-RES-D')
plt.title('贝叶斯校准示例：从含噪观测中估计状态')
plt.legend()
plt.grid(True)
plt.show()

C.7 仿真实验模板

以下是一个完整的PSE实验脚本模板，可用于研究特定干预操作的效果。


def run_intervention_simulation(intervention_schedule, baseline_params, t_max=168):
    """
    intervention_schedule: list of (time, operation, target, intensity)
    """
    t_eval = np.arange(0, t_max, 0.1)
    psi = initial_anxious_state()
    states = [psi.copy()]
    times = [0]
    
    for t_next in t_eval[1:]:
        # 检查干预
        for (t_op, op, target, intensity) in intervention_schedule:
            if times[-1] < t_op <= t_next:
                idx = STATE_INDICES[target]
                if op == 'MODULATE+':
                    psi[idx] = min(100, psi[idx] + intensity)
                elif op == 'GATE-':
                    psi[idx] = max(0, psi[idx] - intensity)
        # 演化一步
        dt = t_next - times[-1]
        stress = external_stress_profile(t_next)
        dpsi = mkt_equations(psi, t_next, baseline_params, stress)
        psi = psi + dpsi * dt
        psi = np.clip(psi, 0, 100)
        states.append(psi.copy())
        times.append(t_next)
    
    return np.array(times), np.array(states)

t, states = run_intervention_simulation(intervention, params, t_max=72)
